get all: curl -X GET http://localhost:8080/users
get by id: curl -X GET http://localhost:8080/users/1
create: curl -X POST http://localhost:8080/users \
     -H "Content-Type: application/json" \
     -d '{"name": "Charlie", "email": "charlie@example.com"}'
update: curl -X PUT http://localhost:8080/users/1 \
     -H "Content-Type: application/json" \
     -d '{"name": "Alice Updated", "email": "alice_new@example.com"}'

delete: curl -X DELETE http://localhost:8080/users/1
